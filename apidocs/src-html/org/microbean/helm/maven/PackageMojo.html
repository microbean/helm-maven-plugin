061/**
062 * Packages a Helm chart from its raw materials.
063 *
064 * @author <a href="https://about.me/lairdnelson"

126  @Override
127  public void execute() throws MojoExecutionException, MojoFailureException {
128    final Log log = this.getLog();
129    assert log != null;
130
131    if (this.getSkip()) {
132      if (log.isDebugEnabled()) {
133        log.debug("Skipping execution by request.");
134      }
135      return;
136    }
137
138    final URI chartContentsUri = this.getChartContentsUri();
139    if (chartContentsUri == null) {
140      throw new IllegalStateException("getChartContentsUri() == null");
141    }
142    URL chartContentsUrl = null;
143    try {
144      chartContentsUrl = chartContentsUri.toURL();
145    } catch (final IOException ioException) {
146      throw new MojoExecutionException(ioException.getMessage(), ioException);
147    }
148    assert chartContentsUrl != null;
149
150    AbstractChartLoader<URL> chartLoader = this.getChartLoader();
151    if (chartLoader == null) {
152      chartLoader = new URLChartLoader();
153    }
154
155    Throwable throwable = null;
156
157    Chart.Builder chart = null;
158    try {
159      chart = chartLoader.load(chartContentsUrl);
160    } catch (final RuntimeException runtimeException) {
161      throwable = runtimeException;
162      throw runtimeException;
163    } catch (final IOException exception) {
164      final MojoExecutionException e = new MojoExecutionException(exception.getMessage(), exception);
165      throwable = e;
166      throw e;
167    } finally {
168      try {
169        chartLoader.close();
170      } catch (final IOException suppressMe) {
171        if (throwable == null) {
172          throw new MojoExecutionException(suppressMe.getMessage(), suppressMe);
173        } else {
174          throwable.addSuppressed(suppressMe);
175        }
176      }
177    }
178    throwable = null;
179
180    final MetadataOrBuilder metadata = chart.getMetadata();
181    if (metadata == null) {
182      throw new IllegalStateException("chart.getMetadata() == null");
183    }
184    final String chartName = metadata.getName();
185    if (chartName == null) {
186      throw new IllegalStateException("metadata.getName() == null");
187    } else if (chartName.isEmpty()) {
188      throw new IllegalStateException("metadata.getName().isEmpty()");
189    }
190
191    AbstractChartWriter chartWriter = this.getChartWriter();
192    if (chartWriter == null) {
193      URI chartTargetUri = this.getChartTargetUri();
194      if (chartTargetUri == null) {
195        final Build build = this.project.getBuild();
196        assert build != null;
197        final String targetDirectory = build.getDirectory();
198        assert targetDirectory != null;
199        final Path targetDirectoryPath = Paths.get(targetDirectory);
200        assert targetDirectoryPath != null;
201        final Path helmChartsDirectoryPath = targetDirectoryPath.resolve("generated-sources/helm/charts");
202        assert helmChartsDirectoryPath != null;
203        assert helmChartsDirectoryPath.isAbsolute();
204        final Path chartDirectoryPath = helmChartsDirectoryPath.resolve(chartName + ".tgz");
205        assert chartDirectoryPath != null;
206        chartTargetUri = chartDirectoryPath.toUri();
207      }
208      assert chartTargetUri != null;
209      if ("file".equals(chartTargetUri.getScheme())) {
210        final String chartTargetUriPath = chartTargetUri.getPath();
211        if (chartTargetUriPath != null) {
212          final Path chartTargetPath = Paths.get(chartTargetUriPath).normalize();
213          assert chartTargetPath != null;
214          final Path parent = chartTargetPath.getParent();
215          if (parent != null) {
216            try {
217              Files.createDirectories(parent);
218            } catch (final IOException ioException) {
219              throw new MojoExecutionException(ioException.getMessage(), ioException);
220            }
221          }
222        }
223      }
224      URL chartTargetURL = null;
225      try {
226        chartTargetURL = chartTargetUri.toURL();
227      } catch (final IOException ioException) {
228        throw new MojoExecutionException(ioException.getMessage(), ioException);
229      }
230      assert chartTargetURL != null;
231      URLConnection connection = null;
232      try {
233        connection = chartTargetURL.openConnection();
234      } catch (final IOException ioException) {
235        throw new MojoExecutionException(ioException.getMessage(), ioException);
236      }      
237      assert connection != null;
238      connection.setDoOutput(true);
239      OutputStream outputStream = null;
240      try {
241        outputStream = connection.getOutputStream();
242      } catch (final UnknownServiceException outputNotSupported) {
243        // http://bugs.java.com/bugdatabase/view_bug.do?bug_id=4191800
244        if ("file".equals(chartTargetURL.getProtocol())) {
245          final Path chartTargetPath = Paths.get(chartTargetURL.getPath());
246          assert chartTargetPath != null;
247          try {
248            outputStream = Files.newOutputStream(chartTargetPath);
249          } catch (final IOException ioException) {
250            ioException.addSuppressed(outputNotSupported);
251            throw new MojoExecutionException(ioException.getMessage(), ioException);
252          }
253        } else {
254          throw new MojoExecutionException(outputNotSupported.getMessage(), outputNotSupported);
255        }
256      } catch (final IOException ioException) {
257        throw new MojoExecutionException(ioException.getMessage(), ioException);
258      }
259      try {
260        chartWriter = new TapeArchiveChartWriter(new BufferedOutputStream(new GZIPOutputStream(outputStream)));
261      } catch (final IOException ioException) {
262        throw new MojoExecutionException(ioException.getMessage(), ioException);
263      }
264    }
265    assert chartWriter != null;
266
267    throwable = null;
268    try {
269      chartWriter.write(chart);
270    } catch (final RuntimeException runtimeException) {
271      throwable = runtimeException;
272      throw runtimeException;
273    } catch (final IOException ioException) {
274      final MojoExecutionException e = new MojoExecutionException(ioException.getMessage(), ioException);
275      throwable = e;
276      throw e;
277    } finally {
278      try {
279        chartWriter.close();
280      } catch (final IOException suppressMe) {
281        if (throwable == null) {
282          throw new MojoExecutionException(suppressMe.getMessage(), suppressMe);
283        } else {
284          throwable.addSuppressed(suppressMe);
285        }
286      }
287    }
288    
289  }
290
291  /**
292   * Implements the {@link Disposable} interface by {@linkplain
293   * AbstractChartLoader#close() closing} the {@link
294   * AbstractChartLoader} returned by the {@link #getChartLoader()}
295   * method and {@linkplain AbstractChartWriter#close() closing} the
296   * {@link AbstractChartWriter} returned by the {@link
297   * #getChartWriter()} method.
298   *
299   * <p>Any {@link IOException}s encountered are {@linkplain
300   * Log#error(Throwable) logged}.</p>
301   */
302  @Override
303  public void dispose() {
304    final Log log = this.getLog();
305    assert log != null;
306
307    final Closeable chartLoader = this.getChartLoader();
308    if (chartLoader != null) {
309      try {
310        chartLoader.close();
311      } catch (final IOException logMe) {
312        if (log.isErrorEnabled()) {
313          log.error(logMe);
314        }
315      }
316    }
317
318    final Closeable chartWriter = this.getChartWriter();
319    if (chartWriter != null) {
320      try {
321        chartWriter.close();
322      } catch (final IOException logMe) {
323        if (log.isErrorEnabled()) {
324          log.error(logMe);
325        }
326      }
327    }
328  }
