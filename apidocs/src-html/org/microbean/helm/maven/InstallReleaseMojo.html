020
021import java.net.URL;
022
023import java.nio.file.Files;
024import java.nio.file.Path;
025import java.nio.file.Paths;
026
027import java.util.Objects;
028
029import java.util.concurrent.Callable;
030import java.util.concurrent.Future;

032import java.util.regex.Matcher;

034import javax.inject.Inject;

036import hapi.chart.ChartOuterClass.Chart;
037
038import hapi.release.ReleaseOuterClass.Release;

045import org.apache.maven.plugin.MojoExecutionException;

047import org.apache.maven.plugin.logging.Log;

049import org.apache.maven.plugins.annotations.Mojo;

052import org.apache.maven.project.MavenProject;

056import org.microbean.helm.chart.AbstractChartLoader;

059@Mojo(name = "install")
060public class InstallReleaseMojo extends AbstractMutatingReleaseMojo {
061
062  private final MavenProject project;
063
064  private final MavenSession session;
065
066  /**
067   * The name of the release to install.  If omitted, a release name
068   * will be generated and used instead.
069   *
070   * @see #getReleaseName()
071   *
072   * @see #setReleaseName(String)
073   *
074   * @see #validateReleaseName(String)
075   */
076  @Parameter
077  private String releaseName;
078  
079  @Parameter
080  private String releaseNamespace;
081
082  @Parameter
083  private boolean reuseReleaseName;
084
085  @Parameter
086  private String valuesYaml;
087
088  @Parameter
089  private URL chartUrl;
090  
091  @Inject
092  public InstallReleaseMojo(final MavenProject project, final MavenSession session) {
093    super();
094    Objects.requireNonNull(project);
095    Objects.requireNonNull(session);
096    this.project = project;
097    this.session = session;
098  }
099
100  @Override
101  protected void execute(final Callable<ReleaseManager> releaseManagerCallable) throws Exception {
102    Objects.requireNonNull(releaseManagerCallable);
103    final Log log = this.getLog();
104    assert log != null;
105
106    URL chartUrl = this.getChartUrl();
107    if (chartUrl == null) {
108      final Path chartDirectoryPath = Paths.get(new StringBuilder(this.project.getBuild().getDirectory()).append("/helm/").append(this.project.getArtifactId()).toString());
109      assert chartDirectoryPath != null;
110      chartUrl = chartDirectoryPath.toUri().toURL();
111      if (!Files.isDirectory(chartDirectoryPath)) {
112        throw new MojoExecutionException("Non-existent chartUrl: " + chartUrl);
113      }
114    }
115    assert chartUrl != null;
116    if (log.isDebugEnabled()) {
117      log.debug("chartUrl: " + chartUrl);
118    }
119
120    Chart.Builder chartBuilder = null;
121    try (final AbstractChartLoader<URL> chartLoader = this.createChartLoader()) {
122      if (chartLoader == null) {
123        throw new IllegalStateException("createChartLoader() == null");
124      }
125      if (log.isDebugEnabled()) {
126        log.debug("chartLoader: " + chartLoader);
127        log.debug("Loading Helm chart from " + chartUrl);
128      }
129      chartBuilder = chartLoader.load(chartUrl);
130    }
131
132    if (chartBuilder == null) {
133      throw new IllegalStateException("chartBuilder.load(\"" + chartUrl + "\") == null");
134    }
135
136    if (log.isInfoEnabled()) {
137      log.info("Loaded Helm chart from " + chartUrl);
138    }
139    
140    final InstallReleaseRequest.Builder requestBuilder = InstallReleaseRequest.newBuilder();
141    assert requestBuilder != null;
142
143    requestBuilder.setDisableHooks(this.getDisableHooks());
144    requestBuilder.setDryRun(this.getDryRun());
145
146    final String releaseName = this.getReleaseName();
147    if (releaseName != null) {
148      requestBuilder.setName(releaseName);
149    }
150
151    final String releaseNamespace = this.getReleaseNamespace();
152    if (releaseNamespace != null) {
153      requestBuilder.setNamespace(releaseNamespace);
154    }
155    
156    requestBuilder.setReuseName(this.getReuseReleaseName());
157    requestBuilder.setTimeout(this.getTimeout());
158
159    final String valuesYaml = this.getValuesYaml();
160    if (valuesYaml != null && !valuesYaml.isEmpty()) {
161      final hapi.chart.ConfigOuterClass.Config.Builder values = requestBuilder.getValuesBuilder();
162      assert values != null;
163      values.setRaw(valuesYaml);
164    }
165
166    requestBuilder.setWait(this.getWait());
167
168    final ReleaseManager releaseManager = releaseManagerCallable.call();
169    if (releaseManager == null) {
170      throw new IllegalStateException("releaseManagerCallable.call() == null");
171    }
172
173    if (log.isInfoEnabled()) {
174      log.info("Installing release " + requestBuilder.getName());
175    }
176    final Future<InstallReleaseResponse> installReleaseResponseFuture = releaseManager.install(requestBuilder, chartBuilder);
177    assert installReleaseResponseFuture != null;
178    final InstallReleaseResponse installReleaseResponse = installReleaseResponseFuture.get();
179    assert installReleaseResponse != null;
180    if (log.isInfoEnabled()) {
181      final Release release = installReleaseResponse.getRelease();
182      assert release != null;
183      log.info("Installed release " + release.getName());
184    }
185    
186  }
187
188  protected AbstractChartLoader<URL> createChartLoader() {
189    return new URLChartLoader();
190  }
191
192  public URL getChartUrl() {
193    return this.chartUrl;
194  }
195
196  public void setChartUrl(final URL chartUrl) {
197    this.chartUrl = chartUrl;
198  }
199  
200  public String getReleaseNamespace() {
201    return this.releaseNamespace;
202  }
203
204  public void setReleaseNamespace(final String releaseNamespace) {
205    this.releaseNamespace = releaseNamespace;
206  }
207
208  public boolean getReuseReleaseName() {
209    return this.reuseReleaseName;
210  }
211
212  public void setReuseReleaseName(final boolean reuseReleaseName) {
213    this.reuseReleaseName = reuseReleaseName;
214  }
215
216  public String getValuesYaml() {
217    return this.valuesYaml;
218  }
219
220  public void setValuesYaml(final String valuesYaml) {
221    this.valuesYaml = valuesYaml;
222  }
223
224  @Override
225  protected void validateReleaseName(final String name) {
226    if (name != null && !name.isEmpty()) {
227      super.validateReleaseName(name);
228    }
229  }
