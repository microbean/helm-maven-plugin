040/**
041 * An {@link AbstractHelmMojo} that provides other <a
042 * href="https://microbean.github.io/microbean-helm/">Helm</a>-related
043 * <a
044 * href="https://maven.apache.org/developers/mojo-api-specification.html">mojo</a>
045 * implementations the ability to work with a {@link ReleaseManager}.
046 *
047 * @author <a href="https://about.me/lairdnelson"
048 * target="_parent">Laird Nelson</a>
049 *
050 * @see #execute(Callable)
051 */
052public abstract class AbstractReleaseMojo extends AbstractHelmMojo {
053
054
055  /*
056   * Instance fields.
057   */
058
059
060  /**
061   * Whether to skip execution.
062   */
063  @Parameter(defaultValue = "false")
064  private boolean skip;
065
066  /**
067   * The <a
068   * href="https://static.javadoc.io/io.fabric8/kubernetes-client/3.1.8/io/fabric8/kubernetes/client/Config.html">{@code
069   * Config}</a> describing how a <a
070   * href="https://static.javadoc.io/io.fabric8/kubernetes-client/3.1.8/io/fabric8/kubernetes/client/DefaultKubernetesClient.html">{@code
071   * DefaultKubernetesClient}</a> should connect to a Kubernetes
072   * cluster.
073   */
074  @Parameter
075  private Config clientConfiguration;
076
077
078  /*
079   * Constructors.
080   */
081
082
083  /**
084   * Creates a new {@link AbstractReleaseMojo}.
085   */
086  protected AbstractReleaseMojo() {
087    super();
088  }
089
090
091  /*
092   * Public instance methods.
093   */
094
095
096  /**
097   * {@linkplain #getSkip() Skips} execution if instructed, or calls
098   * the {@link #execute(Callable)} method with a {@link Callable}
099   * containing the results of an invocation of the {@link
100   * #createReleaseManager(Tiller)} method.
101   *
102   * @exception MojoExecutionException if there was a problem
103   * executing this mojo
104   *
105   * @exception MojoFailureException if the mojo executed properly,
106   * but the job it was to perform failed in some way
107   */
108  @Override
109  public void execute() throws MojoExecutionException, MojoFailureException {
110    final Log log = this.getLog();
111    assert log != null;
112
113    if (this.getSkip()) {
114      if (log.isDebugEnabled()) {
115        log.debug("Skipping execution by request.");
116      }
117      return;
118    }
119
120    final ReleaseManagerCallable releaseManagerCallable = new ReleaseManagerCallable();
121    Throwable throwable = null;
122    try {
123      this.execute(releaseManagerCallable);
124    } catch (final InterruptedException interruptedException) {
125      Thread.currentThread().interrupt();
126      final MojoExecutionException mojoExecutionException = new MojoExecutionException(interruptedException.getMessage(), interruptedException);
127      throwable = mojoExecutionException;
128      throw mojoExecutionException;
129    } catch (final RuntimeException | MojoExecutionException | MojoFailureException throwMe) {
130      throwable = throwMe;
131      throw throwMe;
132    } catch (final Exception otherStuff) {
133      final MojoExecutionException mojoExecutionException = new MojoExecutionException(otherStuff.getMessage(), otherStuff);
134      throwable = mojoExecutionException;
135      throw mojoExecutionException;
136    } finally {
137      if (releaseManagerCallable.releaseManager != null) {
138        try {
139          releaseManagerCallable.releaseManager.close();
140        } catch (final IOException ioException) {
141          if (throwable != null) {
142            throwable.addSuppressed(ioException);
143          } else {
144            throw new MojoExecutionException(ioException.getMessage(), ioException);
145          }
146        }
147      }
148    }
149  }
150
151  /**
152   * Returns the {@link Config} describing how a {@link
153   * DefaultKubernetesClient} is to connect to a Kubernetes cluster.
154   *
155   * <p>This method may return {@code null}.</p>
156   *
157   * <p>Overrides of this method may return {@code null}.</p>
158   *
159   * @return a {@link Config}, or {@code null}
160   *
161   * @see #setClientConfiguration(Config)
162   */
163  public Config getClientConfiguration() {
164    return this.clientConfiguration;
165  }
166
167  /**
168   * Installs the {@link Config} describing how a {@link
169   * DefaultKubernetesClient} is to connect to a Kubernetes cluster.

171   * @param config the {@link Config} to use; may be {@code null}
172   *
173   * @see #getClientConfiguration()
174   */
175  public void setClientConfiguration(final Config config) {
176    this.clientConfiguration = config;
177  }

179  /**
180   * Returns {@code true} if this {@link AbstractReleaseMojo} should
181   * not execute.

183   * @return {@code true} if this {@link AbstractReleaseMojo} should
184   * not execute; {@code false} otherwise
185   *
186   * @see #setSkip(boolean)
187   */
188  public boolean getSkip() {
189    return this.skip;
190  }
191
192  /**
193   * Controls whether this {@link AbstractReleaseMojo} should execute.
194   *
195   * @param skip if {@code true}, this {@link AbstractReleaseMojo}
196   * will not execute
197   *
198   * @see #getSkip()
199   */
200  public void setSkip(final boolean skip) {
201    this.skip = skip;
202  }

210  /**
211   * Performs a release-oriented task using a {@link ReleaseManager}
212   * {@linkplain Callable#call() available} from the supplied {@link
213   * Callable}.
214   *
215   * @param releaseManagerCallable the {@link Callable} that will
216   * provide a {@link ReleaseManager}; must not be {@code null}
217   *
218   * @exception Exception if an error occurs
219   */
220  protected abstract void execute(final Callable<ReleaseManager> releaseManagerCallable) throws Exception;
221
222  /**
223   * Creates a {@link DefaultKubernetesClient} for communicating with
224   * Kubernetes clusters.
225   *
226   * <p>This method never returns {@code null}.</p>
227   *
228   * <p>Overrides of this method must not return {@code null}.</p>
229   *
230   * <p>The default implementation calls the {@link
231   * #getClientConfiguration()} method and {@linkplain
232   * DefaultKubernetesClient#DefaultKubernetesClient(Config) uses its
233   * return value}, unless it is {@code null}, in which case a new
234   * {@link DefaultKubernetesClient} is created via its {@linkplain
235   * DefaultKubernetesClient#DefaultKubernetesClient() no-argument
236   * constructor}.</p>
237   *
238   * @return a new, non-{@code null} {@link DefaultKubernetesClient}
239   *
240   * @exception IOException if there was a problem creating the client
241   */
242  protected DefaultKubernetesClient createClient() throws IOException {
243    final DefaultKubernetesClient client;
244    final Config config = this.getClientConfiguration();
245    if (config == null) {
246      client = new DefaultKubernetesClient();
247    } else {
248      client = new DefaultKubernetesClient(config);
249    }
250    return client;
251  }
252
253  /**
254   * Creates a {@link Tiller} and returns it.
255   *
256   * <p>This method never returns {@code null}.</p>
257   *
258   * <p>Overrides of this method must not return {@code null}.</p>
259   *
260   * <p>This implementation passes the supplied {@link
261   * DefaultKubernetesClient} to the <a
262   * href="https://microbean.github.io/microbean-helm/apidocs/org/microbean/helm/Tiller.html#Tiller-T-">appropriate
263   * <code>Tiller</code> constructor</a>.</p>
264   *
265   * @param client the {@link DefaultKubernetesClient} to use to
266   * communicate with a Kubernetes cluster; must not be {@code null}
267   *
268   * @return a new {@link Tiller}; never {@code null}
269   *
270   * @exception NullPointerException if {@code client} is {@code null}
271   *
272   * @exception IOException if there was a problem creating a {@link
273   * Tiller}
274   */
275  protected Tiller createTiller(final DefaultKubernetesClient client) throws IOException {
276    Objects.requireNonNull(client);
277    return new Tiller(client);
278  }
279
280  /**
281   * Creates a {@link ReleaseManager} and returns it.
282   *
283   * <p>This method never returns {@code null}.</p>
284   *
285   * <p>Overrides of this method must not return {@code null}.</p>
286   *
287   * <p>This implementation passes the supplied {@link
288   * Tiller} to the {@linkplain ReleaseManager#ReleaseManager(Tiller)
289   * appropriate <code>ReleaseManager</code> constructor}.</p>
290   *
291   * @param tiller the {@link Tiller} to use to communicate with a
292   * Tiller server; must not be {@code null}
293   *
294   * @return a new {@link ReleaseManager}; never {@code null}
295   *
296   * @exception NullPointerException if {@code tiller} is {@code null}
297   *
298   * @exception IOException if there was a problem creating a {@link
299   * ReleaseManager}
300   */
301  protected ReleaseManager createReleaseManager(final Tiller tiller) throws IOException {
302    Objects.requireNonNull(tiller);
303    return new ReleaseManager(tiller);
304  }
305
306  /**
307   * Validates a <a
308   * href="https://kubernetes.io/docs/concepts/overview/working-with-objects/namespaces/">Kubernetes
309   * namespace</a> for correctness.
310   *
311   * <p>The default implementation checks the supplied {@code
312   * namespace} to see if it is less than or equal to {@value
313   * ReleaseManager#DNS_LABEL_MAX_LENGTH} characters, and if it
314   * {@linkplain Matcher#matches() matches} the value of the {@link
315   * ReleaseManager#DNS_LABEL_PATTERN} field.</p>
316   *
317   * @param namespace the <a
318   * href="https://kubernetes.io/docs/concepts/overview/working-with-objects/namespaces/">namespace</a>
319   * to validate; may be {@code null}
320   *
321   * @exception IllegalArgumentException if {@code namespace} is
322   * invalid
323   */
324  protected void validateNamespace(final String namespace) {
325    if (namespace != null) {
326      final int namespaceLength = namespace.length();
327      if (namespaceLength > ReleaseManager.DNS_LABEL_MAX_LENGTH) {
328        throw new IllegalArgumentException("Invalid namespace: " + namespace + "; length is greater than " + ReleaseManager.DNS_LABEL_MAX_LENGTH + " characters: " + namespaceLength);
329      } else if (namespaceLength > 0) {
330        final Matcher matcher = ReleaseManager.DNS_LABEL_PATTERN.matcher(namespace);
331        assert matcher != null;
332        if (!matcher.matches()) {
333          throw new IllegalArgumentException("Invalid namespace: " + namespace + "; must match " + ReleaseManager.DNS_LABEL_PATTERN.toString());
334        }
335      }
336    }
337  }

341   * Inner and nested classes.
342   */

344
345  /**
346   * A {@link Callable} whose {@link #call()} method yields the same
347   * {@link ReleaseManager} for every invocation, {@linkplain
348   * AbstractReleaseMojo#createReleaseManager(Tiller) creating one} if
349   * necessary.
350   *
351   * @author <a href="https://about.me/lairdnelson"
352   * target="_parent">Laird Nelson</a>
353   *
354   * @see AbstractReleaseMojo#createReleaseManager(Tiller)
355   *
356   * @see AbstractReleaseMojo#createTiller(DefaultKubernetesClient)
357   *
358   * @see AbstractReleaseMojo#createClient()
359   */
360  private final class ReleaseManagerCallable implements Callable<ReleaseManager> {
361
362
363    /*
364     * Instance fields.
365     */
366
367
368    /**
369     * The {@link ReleaseManager} to return from the {@link #call()}
370     * method.
371     *
372     * <p>This field may be {@code null}.</p>
373     *
374     * @sed #call()
375     */
376    private ReleaseManager releaseManager;
377
378
379    /*
380     * Constructors.
381     */
382
383
384    /**
385     * Creates a new {@link ReleaseManagerCallable}.
386     */
387    private ReleaseManagerCallable() {
388      super();
389    }
390
391
392    /*
393     * Instance methods.
394     */
395
396
397    /**
398     * Returns a {@link ReleaseManager}, {@linkplain
399     * AbstractReleaseMojo#createReleaseManager(Tiller) creating one}
400     * if necessary.
401     *
402     * <p>This method never returns {@code null}.</p>
403     *
404     * @return a {@link ReleaseManager}; never {@code null}
405     *
406     * @exception IOException if there was a problem creating a {@link
407     * ReleaseManager}
408     */
409    @Override
410    public final ReleaseManager call() throws IOException {
411      if (this.releaseManager == null) {
412        this.releaseManager = createReleaseManager(createTiller(createClient()));
413      }
414      return this.releaseManager;
415    }
416    
417  }
